**Step 1: Create `frontend/src/app/services/logger.service.ts`**

```typescript
import { Injectable } from '@angular/core';
import { environment } from '../../environments/environment';

@Injectable({ providedIn: 'root' })
export class LoggerService {
  log(message: string, ...args: unknown[]): void {
    if (!environment.production) {
      console.log(message, ...args);
    }
  }

  warn(message: string, ...args: unknown[]): void {
    if (!environment.production) {
      console.warn(message, ...args);
    }
  }

  error(message: string, ...args: unknown[]): void {
    if (!environment.production) {
      console.error(message, ...args);
    }
  }
}
```

This is the simplest valid design per RESEARCH.md Pattern 4. Three methods: `log`, `warn`, `error`. Each checks `environment.production` and is a no-op when true.
